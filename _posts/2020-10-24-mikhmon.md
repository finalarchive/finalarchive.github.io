## Install

  sudo su
  apt update
  apt -y install curl git php
  
  sudo su
  cd /usr/local/bin
  mkdir mikhmon
  cd mikhmon
  curl -o install-mikhmon https://laksa19.github.io/install-mikhmon.txt

edit

  nano install-mikhmon
  
  #bin/bash
  #pkg install git php -y
  
 chmod +x install-mikhmon
 ./install-mikhmon
